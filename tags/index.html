// 只要是没点赞的，就自动点赞，每隔300多毫秒，点一下，当然得把它的文章滚动到最下方。
[...document.querySelectorAll('.item.like')].filter(item => item.className !== 'item like active').forEach((item, i) => {
    setTimeout(()=>{
        item.click()
    }, i * 300 + (~~ (Math.random() * 10)))
})
